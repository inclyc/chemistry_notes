\chapter*{学期过渡部分}
    \section*{基本反应回顾}
    \begin{itemize}
        \item 烷烃--自由基引发，取代
        \item 烯烃--加成，亲电，$\pi$电子云是裸露在外的
        \item P--X 卤代烃 没有不饱和度，被亲核试剂进攻
        \item 芳烃 亲电试剂进攻
    \end{itemize}

    \section*{亲核试剂和亲电试剂}
    
    \subsection*{亲电试剂}

    \begin{enumerate}
        \item 单质 $\ce{X2}$
        \item 强酸 $\ce{HCl}$
        \item 缺电子化物 
    \end{enumerate}

    \subsection*{亲核试剂}

    \begin{enumerate}
        \item 弱酸、弱酸眼 $\ce{HCN}$，$\ce{NaCN}$
        \item 含有未成键电子对化合物 $\ce{NH3}$
        \item 格氏试剂 $\ce{RMgX}$
    \end{enumerate}
